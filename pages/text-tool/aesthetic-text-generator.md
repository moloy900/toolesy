    function generateFontStyles() {
      aestheticTextList.innerHTML = '';
      
      fontStyles.forEach((style, index) => {
        const fontDiv = document.createElement('div');
        fontDiv.className = 'singular-font';
        fontDiv.setAttribute('data-font', style.name);
        fontDiv.innerHTML = `
          <span class="aesthetic-text-output" data-index="${index}">${style.name} - Enter text to see preview</span>
          <button class="copy-text-btn" data-index="${index}">
            <i class="fas fa-copy"></i> Copy
          </button>
        `;
        aestheticTextList.appendChild(fontDiv);
        
        // Add copy event listener
        const copyBtn = fontDiv.querySelector('.copy-text-btn');
        copyBtn.addEventListener('click', function() {
          const textIndex = this.getAttribute('data-index');

        });
      });

    function convertText() {
      const text = textInput.value;
      if (!text) {
        fontPreview.textContent = 'Your aesthetic text will appear here...';
        updateFontPreviews('Enter text to see previews');
        return;
      }

      // Update main preview
      const previewText = fontStyles[0].converter(text);
      fontPreview.textContent = previewText || 'Your aesthetic text will appear here...';

      // Update all font style previews
      updateFontPreviews(text);
    }

    function updateFontPreviews(text) {
      const outputElements = document.querySelectorAll('.aesthetic-text-output');
      outputElements.forEach((element, index) => {
        if (index < fontStyles.length) {
          const convertedText = fontStyles[index].converter(text);
          element.textContent = convertedText || `${fontStyles[index].name} - Enter text to see preview`;
        }
      });
    }

    function handleTextAction(action) {
      switch (action) {
        case 'copyAll':
          const allTexts = fontStyles.map(style => `${style.name}:\n${style.converter(textInput.value)}`).join('\n\n');
          copyToClipboard(allTexts, 'All Styles');
          break;

        case 'clear':
          textInput.value = '';
          fontPreview.textContent = 'Your aesthetic text will appear here...';
          updateCounts();
          updateFontPreviews('Enter text to see previews');
          showAlert('Text cleared successfully!', 'success');
          break;

        case 'reverse':
          const reversedText = textInput.value.split('').reverse().join('');
          textInput.value = reversedText;
          convertText();
          showAlert('Text reversed successfully!', 'success');
          break;

        case 'refresh':
          convertText();
          showAlert('Styles refreshed!', 'success');
          break;
      }
    }

    function copyToClipboard(text, styleName) {
      if (!text || text.includes('Enter text to see preview')) {
        showAlert('Please generate some text first!', 'error');
        return;
      }
      
      navigator.clipboard.writeText(text).then(() => {
        showAlert(`"${styleName}" text copied to clipboard!`, 'success');
      }).catch(err => {
        showAlert('Failed to copy text: ' + err, 'error');
      });
    }

    function showAlert(message, type) {

      alertDiv.className = `alert alert-${type}`;
      alertDiv.innerHTML = `
        ${message}
        <span class="close">&times;</span>
      `;

      alertContainer.innerHTML = '';
      alertContainer.appendChild(alertDiv);

      // Add close functionality

        alertDiv.remove();
      });

      setTimeout(() => {
        if (alertDiv.parentNode) {
          alertDiv.remove();
        }
      }, 5000);
